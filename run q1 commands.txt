Assignment 1 – Hadoop MapReduce run commands (WSL Ubuntu / Linux)

=== Build (run once) ===
cd "/mnt/g/Uni/Big Data/Assignment1"
mvn -DskipTests package
# Produces 5 JARs under target/:
#   InnerJoin.jar  FullOuterJoin.jar  SetDifference.jar  FriendsOfFriends.jar  HadoopSort.jar

=== Q1.1 – Inner Join ===
rm -rf out_inner
mvn -DskipTests exec:java -Dexec.mainClass=assignment1.InnerJoin -Dexec.args="input_q1.txt out_inner"
cat out_inner/part-r-00000
# Expected:
#   A    1,x
#   B    2,x
#   C    3,x

=== Q1.2 – Full Outer Join ===
rm -rf out_outer
mvn -DskipTests exec:java -Dexec.mainClass=assignment1.FullOuterJoin -Dexec.args="input_q1.txt out_outer"
cat out_outer/part-r-00000
# Expected:
#   A      1,x
#   B      2,x
#   C      3,x
#   null   4,y
#   null   5,y
#   null   6,y
#   null   7,z
#   null   8,z

=== Q1.3 – Set Difference A1[T1] – A1[T2] ===
rm -rf out_diff
mvn -DskipTests exec:java -Dexec.mainClass=assignment1.SetDifference -Dexec.args="input_q1.txt out_diff"
cat out_diff/part-r-00000
# Expected: 4  5  6  7  8

=== Q2 – Friends of Friends (all pairs) ===
rm -rf out_fof
mvn -DskipTests exec:java -Dexec.mainClass=assignment1.FriendsOfFriends -Dexec.args="input_q2.txt out_fof"
cat out_fof/part-r-00000

=== Q2 – Friends of Friends (filtered for P1) ===
rm -rf out_fof_p1 out_fof_p1_temp
mvn -DskipTests exec:java -Dexec.mainClass=assignment1.FriendsOfFriends -Dexec.args="input_q2.txt out_fof_p1 P1"
cat out_fof_p1/part-r-00000

=== Q3 – Hadoop Sort ===
rm -rf out_sort
mvn -DskipTests exec:java -Dexec.mainClass=assignment1.HadoopSort -Dexec.args="input_q3.txt out_sort"
cat out_sort/part-r-00000
# Keys are sorted ascending; edit input_q3.txt with any key<TAB>value pairs.

=== Notes ===
- Always delete the output folder before rerunning the same job.
- Intermediate temp folders (e.g. out_fof_temp) are created automatically; delete them too on rerun.
- All paths are relative to the project root.
- For actual HDFS usage replace local paths with hdfs:// URIs and remove the
  mapreduce.framework.name=local / fs.defaultFS=file:/// configuration overrides.
